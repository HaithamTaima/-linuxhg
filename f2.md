```c
#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {

    int n;
    n = fork();

    if (n == -1) {
        printf("stderr, fork has failed \n");
        exit(1);
    }
    else if (n == 0) {

        printf("the son process, fork value = %d \n", n);
        printf("process son identification : %d \n", getpid());
        printf("parent process identification : %d \n", getppid());
        printf("end of the son process \n");
    }
    else {

        printf("in the parent process, fork value = %d \n", n);
        printf("parent process identification : %d \n", getpid());
        printf("parent process identification of parent process : %d \n", getppid());
        printf("end of the parent process \n");
    }

    return 0;
}
```

```c
{
    if (fork() == 0)
    {
        printf("son process pid %d\n", getpid());
        sleep(5);
        printf("son will exit(2)\n");
        exit(2);
    }
    else
    {
        int m, n;
        printf("parent will wait\n");
        m = wait(&n);
        printf("parent: %d end son process %d with return value of wait %d\n",
               getpid(), m, WEXITSTATUS(n));
    }

    return 0;
}
```
